← Dashboard Canvas Emails
One-time setup — about 2 minutes
This uses a Google Apps Script that runs in your own Google account to read your Canvas emails. No third-party servers or API permissions needed.
1
Open script.google.com, click New project, and replace all the code with this:
function doGet(e) { var threads = GmailApp.search('from:instructure.com', 0, 50); var emails = []; for (var i = 0; i < threads.length; i++) { var msgs = threads[i].getMessages(); for (var j = 0; j < msgs.length; j++) { var m = msgs[j]; var body = m.getPlainBody().split('________')[0].trim(); emails.push({ id: m.getId(), subject: m.getSubject(), from: m.getFrom(), date: m.getDate().getTime(), body: body.substring(0, 600), unread: m.isUnread() }); } } emails.sort(function(a,b){ return b.date - a.date; }); var json = JSON.stringify({emails: emails}); var cb = e && e.parameter && e.parameter.callback; if (cb) { return ContentService .createTextOutput(cb + '(' + json + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
2
Click Deploy → New deployment. Set type to Web app, Execute as: Me, Who has access: Anyone with Google account. Click Deploy and authorize when prompted.
3
Copy the Web app URL (ends in /exec) and paste it below.
When you click Refresh, a small popup window will open briefly to fetch your emails using your Google session, then close automatically. Make sure popups are allowed for this site.
