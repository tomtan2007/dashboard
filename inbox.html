One-time setup
Same pattern as the Canvas emails page — an Apps Script syncs your inbox into a Google Sheet, which is published as a public CSV this page reads.
1
Create a new Google Sheet. Copy the spreadsheet ID from the URL (between /d/ and /edit).
2
Open script.google.com → New project → paste this code. Replace YOUR_SHEET_ID.
function syncInbox() { var threads = GmailApp.search('in:inbox newer_than:14d', 0, 200); var emails = []; for (var i = 0; i < threads.length; i++) { var msgs = threads[i].getMessages(); var m = msgs[msgs.length - 1]; var body = m.getPlainBody(); if (body) body = body.split('________')[0].trim(); else body = ''; emails.push([ m.getId(), m.getSubject(), m.getFrom(), m.getDate().getTime(), body.substring(0, 800), m.isUnread() ]); } emails.sort(function(a, b) { return b[3] - a[3]; }); var ss = SpreadsheetApp.openById('YOUR_SHEET_ID'); var sheet = ss.getSheetByName('inbox'); if (!sheet) sheet = ss.insertSheet('inbox'); sheet.clearContents(); sheet.getRange(1, 1, 1, 6).setValues( [['id','subject','from','date','body','unread']] ); if (emails.length > 0) { sheet.getRange(2,1,emails.length,6).setValues(emails); } }
3
Click ▶ Run, authorize when prompted. Then set up a time trigger (clock icon → Add Trigger → syncInbox → every 15 min).
4
In the Sheet: File → Share → Publish to web → select the inbox sheet → CSV → Publish. Copy the URL.
If you already set up the Canvas emails sheet, you can reuse the same spreadsheet — the script creates its own "inbox" tab.
